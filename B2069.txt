#include<stdio.h>

int main(void) {
    int n;
    scanf("%d", &n);
    int p[35] = { 0,1 }, q[35] = {0,2};//在题目给定一个范围之后，可以适当地将数组多开一些空间，比如5个，这样可以有效防止数组越界
    //这里的赋值是什么含义复习一下，只有0才是全体赋值
    //对于数列的题目，一般可以选择和题目要求一样对齐下标
    double sum = 0;
    for (int i = 1; i <= n; i++) {//注意这里要写i<=n，因为上面我们要求对齐下标
        q[i + 1] = q[i] + p[i];
        p[i + 1] = q[i];
        sum += 1.0 * q[i] / p[i];//每一步直接计算
    }//这里我们可以发现我们只需要n项，但是计算到了n+1项，如果一开始数组不开大一点就会越界报错
    printf("%.4lf", sum);  
}
