#include<stdio.h>

int main(void) {
    int n;
    scanf_s("%d", &n);
    int x, a[4] = {0};//个十百千
    int ans = 0;//存储符合条件的数
    for (int i = 0; i < n; i++) {
        scanf_s("%d", &x);
        for (int j = 0; j < 4; j++) {//注意这里的循环变量必须要换个字母，不能用i，i写习惯了很容易犯这个错误
            a[j] = x % 10;
            x /= 10;
        }
        if (a[0] - a[3] - a[2] - a[1] > 0)
            ans++;
    }
    printf("%d", ans);
}
//这种题还可以把四位数转化成一个字符串，然后个十百千位就可以通过s[0]这样的方式取了，再使用-'0'由ASCII码转化成对应数字就可以了
